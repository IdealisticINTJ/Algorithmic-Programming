import java.util.*;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        double mealCost = sc.nextDouble(); 
        int tip  = sc.nextInt();    
        int tax  = sc.nextInt();    
        int totalCost = (int) Math.round(mealCost + 
                                         mealCost * tip / 100.0 + 
                                         mealCost * tax / 100.0);
        
        System.out.println("The total meal cost is " + totalCost + " dollars.");
    }
}
